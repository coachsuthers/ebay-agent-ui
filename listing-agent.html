Google Sheets & Drive setup guide

This script handles both item data logging to Sheets and photo storage to Drive. Update your existing Apps Script by replacing all the code with the version below, then follow the authorisation steps carefully.

  1. Go to script.google.com and open your existing project
  2. Select all the code and replace it with the script below
  3. Authorise Drive access: in the function dropdown (top toolbar) select authoriseDrive → click Run → click through all Google permission prompts and grant access to both Sheets and Drive
  4. Click Deploy → Manage deployments → pencil icon → New version → Deploy
  5. Test: visit https://your-railway-url.up.railway.app/test/drive — should return {"ok":true}

📚 Drive folder structure: My Drive → eBay Listings → 2026-06-28 — Nike Air Max 90 → photos. Created automatically. Does not need to be in the same location as the spreadsheet.

// eBay Agent — Apps Script (replace all existing code with this)
var SHEET_NAME = "Items";
var ROOT_FOLDER = "eBay Listings"; // top-level folder in My Drive
 
// ── Run this function ONCE to authorise Drive access ─────────────
// Select "authoriseDrive" in the dropdown above and click Run.
// Click through ALL Google permission prompts. Do this before deploying.
function authoriseDrive() {
  // Call every DriveApp method we use — forces Google to request all permissions
  var root = DriveApp.getRootFolder();
  var testFolder = root.createFolder("_ebay_agent_auth_test");
  var testBlob = Utilities.newBlob("test", "text/plain", "test.txt");
  var testFile = testFolder.createFile(testBlob);
  testFile.setTrashed(true);
  testFolder.setTrashed(true);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  Logger.log("All permissions granted: Drive read+write, Sheets. You can now deploy.");
}
 
// Get or create a folder by name inside a parent folder
function getOrCreateSubfolder(parent, name) {
  var folders = parent.getFoldersByName(name);
  return folders.hasNext() ? folders.next() : parent.createFolder(name);
}
 
// Get or create the root eBay Listings folder
function getRootFolder() {
  var folders = DriveApp.getFoldersByName(ROOT_FOLDER);
  return folders.hasNext() ? folders.next() : DriveApp.createFolder(ROOT_FOLDER);
}
 
function doPost(e) {
  try {
    var raw = e.postData ? e.postData.contents : "{}";
    var data = JSON.parse(raw);
 
    // ── Batch image upload ────────────────────────
    if (data.action === "uploadBatch") {
      var root = getRootFolder();
      var folder = getOrCreateSubfolder(root, data.folderName || "Uploads");
      var results = [];
      for (var i = 0; i < data.images.length; i++) {
        var img = data.images[i];
        var bytes = Utilities.base64Decode(img.base64);
        var blob = Utilities.newBlob(bytes, img.mimeType || "image/jpeg", img.filename || ("photo_" + (i+1) + ".jpg"));
        var file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        results.push({url: "https://drive.google.com/uc?export=view&id=" + file.getId(), id: file.getId()});
      }
      return ContentService.createTextOutput(
        JSON.stringify({status:"ok", results:results, folderId:folder.getId()}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    // ── Single image upload ───────────────────────
    if (data.action === "uploadImage") {
      var root = getRootFolder();
      // Use item subfolder name sent from agent, or fall back to date
      var subName = data.folderName || Utilities.formatDate(new Date(), "Europe/London", "yyyy-MM-dd");
      var folder = getOrCreateSubfolder(root, subName);
      var bytes = Utilities.base64Decode(data.base64);
      var blob = Utilities.newBlob(bytes, data.mimeType || "image/jpeg", data.filename || "photo.jpg");
      var file = folder.createFile(blob);
      // Share with link so eBay can access the image URL
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      var url = "https://drive.google.com/uc?export=view&id=" + file.getId();
      return ContentService.createTextOutput(
        JSON.stringify({status:"ok", url:url, id:file.getId(), folderId:folder.getId()}))
        .setMimeType(ContentService.MimeType.JSON);
    }
 
    // ── Item log ──────────────────────────────────
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["Date","Name","Brand","Model","Category","Subcategory",
        "Colour","Material","Size","Gender","Era","Country","Key Specs","Condition",
        "eBay Sold Avg","eBay Live Avg","Vinted Avg","Retail Price",
        "Recommended Price","Best Offer Floor","Price Confidence",
        "Listing Title","eBay Condition","Category Path","Postage","Photo Folder"]);
    }
    var folderUrl = data.photoFolderId
      ? "https://drive.google.com/drive/folders/" + data.photoFolderId : "";
    sheet.appendRow([
      new Date(), data.name, data.brand, data.model,
      data.category, data.subcategory, data.color, data.material,
      data.size, data.gender, data.approximate_age, data.country_of_manufacture,
      data.key_specs, data.condition_summary,
      data.ebay_sold_avg, data.ebay_current_avg,
      data.vinted_avg, data.retail_price, data.recommended_price,
      data.best_offer_floor, data.price_confidence,
      data.listing_title, data.ebay_condition, data.category_path, data.postage,
      folderUrl]);
    return ContentService.createTextOutput(JSON.stringify({status:"ok"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:"error",message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
 
function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({status:"ok",message:"eBay Agent connected"}))
    .setMimeType(ContentService.MimeType.JSON);
}
